import os from flask import Flask, render_template from dotenv import load_dotenv # Load environment variables from .env file load_dotenv() app = Flask(__name__, template_folder='templates') @app.route('/') def home(): # Retrieve the API key from the environment variables gemini_api_key = os.getenv("GOOGLE_API_KEY") # Pass the API key to the HTML template return render_template('recreated_ui.html', gemini_api_key=gemini_api_key) if __name__ == '__main__': # Use environment variable for port if available, otherwise default to 5000 port = int(os.environ.get('PORT', 5000)) app.run(host='0.0.0.0', port=port)
